ﻻ ﻳﻤﻜﻨﻨــﺎ اﻟﺘﻨﺒــﺆ ﺑﺎﻷوﻗــﺎت اﻟﺘــﻲ ﻳﻤﻜــﻦ أن ﺗﺤــﺪث ﺧﻼﻟﻬــﺎ ﻣﺸــﻜﻼت ﺗﺘﻌﻠــﻖ ﺑﻤﻨﺘﺠﺎﺗﻨــﺎ‪ .‬وﺑﻨــﺎءً ﻋﻠــﻰ ذﻟــﻚ‪ ،‬ﻓــﺈن‬ ‫ﻣﺴــﺆوﻟﻴﺘﻨﺎ ﺗﻘﺘﺼــﺮ ﻋﻠــﻰ اﻟﺤــﺪ اﻷﻗﺼــﻰ اﻟــﺬي ﺗﺴــﻤﺢ ﺑــﻪ اﻟﻘﻮاﻧﻴــﻦ اﻟﺴــﺎرﻳﺔ‪ ،‬وﻟــﻦ ﻧﺘﺤﻤــﻞ أي ﻣﺴــﺆوﻟﻴﺔ ﺗﺤــﺖ أي‬ ‫ﻇــﺮف ﻣــﻦ اﻟﻈــﺮوف ﺗﺠــﺎه أي ﺧﺴــﺎﺋﺮ ﻓــﻲ اﻷرﺑــﺎح أو اﻹﻳــﺮادات أو اﻟﻤﻌﻠﻮﻣــﺎت أو اﻟﺒﻴﺎﻧــﺎت أو أي أﺿــﺮار ﺗﺒﻌﻴــﺔ أو‬ ‫ﺧﺎﺻــﺔ أو ﻏﻴــﺮ ﻣﺒﺎﺷــﺮة أو ﺗﺤﺬﻳﺮﻳــﺔ أو ﻋﻘﺎﺑﻴــﺔ أو ﻋﺎرﺿــﺔ ﺗﻨﺸــﺄ ﻋــﻦ أو ﺗﺮﺗﺒــﻂ ﺑﻬــﺬه اﻟﺸــﺮوط أو ﻣﻨﺘﺠــﺎت‬ ‫اﻟﺤﻤﺎﻳﻞ‪.‬‬

٨‬- اﻟﻨﺰاﻋﺎت‬

‫ﻧﺴــﻌﻰ ﺟﺎﻫﺪﻳــﻦ إﻟــﻰ ﺗﻮﻓﻴــﺮ ﻗﻮاﻋــﺪ واﺿﺤــﺔ ﻟﻠﺤــﺪ ﻣــﻦ ﻧﺸــﻮب ﻧﺰاﻋــﺎت ﺑﻴﻨــﻚ وﺑﻴﻨﻨــﺎ‪ ،‬ﺑــﻞ وﻣﻨﻌﻬــﺎ‬ ‫أﻣﻜــﻦ‪ ،‬وﻓــﻲ ﺣﺎﻟــﺔ ﻧﺸــﻮب ﻧــﺰاع‪ ،‬ﻋﻠــﻰ اﻟﺮﻏــﻢ ﻣــﻦ ذﻟــﻚ‪ ،‬ﺳﺘﺴــﺮي ﻗﻮاﻧﻴــﻦ ﻣﻤﻠﻜــﺔ اﻟﺒﺤﺮﻳــﻦ ﻋﻠــﻰ أي ﻣﻄﺎﻟﺒــﺔ أو‬ ‫ﺳــﺒﺐ دﻋــﻮى أو ﻧــﺰاع ﻟﺪﻳــﻚ ﺿﺪﻧــﺎ ﻳﻨﺸــﺄ ﻋــﻦ أو ﻳﺘﻌﻠــﻖ ﺑﻬــﺬه اﻟﺸــﺮوط أو اﻟﻤﻨﺘﺠــﺎت ("اﻟﻤﻄﺎﻟﺒــﺔ") ‪ ،‬وﻳﺠــﻮز ﻟــﻚ‬ ‫ﺣﻞ ﻣﻄﺎﻟﺒﺘﻚ ﻓﻲ ﻣﺤﺎﻛﻢ ﻣﻤﻠﻜﺔ اﻟﺒﺤﺮﻳﻦ‪.‬‬


‫ﻓــﻲ ﺣﺎﻟــﺔ ﺗﻌـّـﺬر ﺗﻨﻔﻴــﺬ أي ﺟــﺰء ﻣــﻦ ﻫــﺬه اﻟﺸــﺮوط‪ ،‬ﻓــﺈن اﻟﺠــﺰء اﻟﻤﺘﺒﻘــﻲ ﻳﻈــﻞ ﻧﺎﻓـ ً‬ ‫ـﺬا وﺳــﺎرﻳً ﺎ ﺑﺎﻟﻜﺎﻣــﻞ‪ ،‬إذا ﻋﺠﺰﻧــﺎ‬ ‫ﻋــﻦ ﺗﻨﻔﻴــﺬ أي ﺷــﺮط ﻣــﻦ ﻫــﺬه اﻟﺸــﺮوط‪ ،‬ﻓــﻼ ﻳﻌــﺪ ﻫــﺬا ﺗﻨـ ً‬ ‫ـﺎزﻻ ﻋﻨــﻪ‪ ،‬أي ﺗﻌﺪﻳــﻞ أو ﺗﻨــﺎزل ﻋــﻦ ﻫــﺬه اﻟﺸــﺮوط‬ ‫وﻣﻮﻗﻌﺎ ﺑﻮاﺳﻄﺘﻨﺎ ‪.‬‬ ‫ً‬ﻳﺠﺐ أن ﻳﻜﻮن ﻣﻜﺘﻮﺑً ﺎ‬ ‫ﻻ ﻳﺤــﻖ ﻟــﻚ ﻧﻘــﻞ أي ﻣــﻦ ﺣﻘﻮﻗــﻚ أو واﺟﺒﺎﺗــﻚ ﺑﻤﻮﺟــﺐ ﻫــﺬه اﻟﺸــﺮوط إﻟــﻰ أي ﺷــﺨﺺ آﺧــﺮ ﻣــﻦ دون اﻟﺤﺼــﻮل‬ ‫ﻋﻠﻰ ﻣﻮاﻓﻘﺘﻨﺎ ‪.‬‬

٩‬- أﺣﻜﺎم إﺿﺎﻓﻴﺔ‬

أﻧﺖ ﺗﻮاﻓﻖ ﻋﻠﻰ اﻻﻟﺘﺰام ﺑﺸﺮوط اﻟﺪﻓﻊ واﻻﺷﺘﺮاﻛﺎت اﻟﺘﻲ ﺗﻮﺟﺪ ﻓﻲ‬ ‫ﻟﻦ ﺗﻘﻮم ﺑﻨﻘﻞ أي ﻣﻦ ﺣﻘﻮﻗﻚ أو اﻟﺘﺰاﻣﺎﺗﻚ ﺑﻤﻮﺟﺐ ﻫﺬه اﻟﺸﺮوط إﻟﻰ أي ﺷﺨﺺ آﺧﺮ دون ﻣﻮاﻓﻘﺘﻨﺎ‪.‬‬ ‫ﻻ ﺗﻤﻨــﺢ ﻫــﺬه اﻟﺸــﺮوط أي ﺣﻘــﻮق اﻧﺘﻔــﺎع ﻟﻄــﺮف ﺛﺎﻟــﺚ‪ّ ،‬‬ ‫إن ﺟﻤﻴــﻊ اﻟﺤﻘــﻮق واﻟﻮاﺟﺒــﺎت اﻟﻤﻜﻔﻮﻟــﺔ ﺑﻤﻮﺟــﺐ ﻫــﺬه‬ ‫اﻟﺸــﺮوط ﻗﺎﺑﻠــﺔ ﻟﻠﺘﺨﺼﻴــﺺ ﺑﺤﺮﻳــﺔ ﻣــﻦ ﻗﺒﻠﻨــﺎ ﻓﻴﻤــﺎ ﻳﺘﻌﻠــﻖ ﺑﺪﻣــﺞ اﻷﺻــﻮل أو ﺷــﺮاﺋﻬﺎ أو ﺑﻴﻌﻬــﺎ أو ﺑﻮاﺳــﻄﺔ‬ ‫ﺗﻨﻔﻴﺬ اﻟﻘﺎﻧﻮن أو ﺧﻼﻓﻪ ‪.‬‬


‫إذا ﺗﻐﻴــﺮت ﻣﻠﻜﻴــﺔ أو اﻟﺘﺤﻜــﻢ ﻓــﻲ ﻛﻞ ﻣﻨﺘﺠﺎﺗﻨــﺎ أو ﺟــﺰء ﻣﻨﻬــﺎ أو أﺻﻮﻟﻬــﺎ ‪ ،‬ﻓﻘــﺪ ﻧﻨﻘــﻞ ﻣﻌﻠﻮﻣﺎﺗــﻚ إﻟــﻰ اﻟﻤﺎﻟــﻚ‬ ‫اﻟﺠﺪﻳﺪ ﺑﻨﻔﺲ اﻟﺸﺮوط واﻷﺣﻜﺎم‪.‬‬


ﻳﺠــﺐ أن ﺗﻌﻠــﻢ أﻧﻨــﺎ ﻗــﺪ ﻧﻀﻄــﺮ إﻟــﻰ ﺗﻐﻴﻴــﺮ اﺳــﻢ اﻟﻤﺴــﺘﺨﺪم اﻟﺨــﺎص ﺑﺤﺴــﺎﺑﻚ ﻓــﻲ ﻇــﺮوف ﻣﻌﻴﻨــﺔ (ﻋﻠــﻰ ﺳــﺒﻴﻞ‬ ‫اﻟﻤﺜــﺎل‪ ،‬إذا ﻃﺎﻟــﺐ ﺷــﺨﺺ آﺧــﺮ ﺑﺎﻟﺤﺼــﻮل ﻋﻠــﻰ اﺳــﻢ اﻟﻤﺴــﺘﺨﺪم ﻫــﺬا وﺛﺒــﺖ أﻧــﻪ ﻻ ﻳﺮﺗﺒــﻂ ﺑﺎﺳــﻤﻚ اﻟــﺬي‬ ‫ﺗﺴﺘﺨﺪﻣﻪ ﻓﻲ ﺣﻴﺎﺗﻚ اﻟﻴﻮﻣﻴﺔ) ‪.


داﺋﻤــﺎ ﻣﻼﺣﻈﺎﺗــﻚ واﻗﺘﺮاﺣﺎﺗــﻚ ﺑﺸــﺄن ﻣﻨﺘﺠﺎﺗﻨــﺎ وﺧﺪﻣﺎﺗﻨــﺎ‪ ،‬وﻟﻜــﻦ ﻳﻨﺒﻐــﻲ ﻋﻠﻴــﻚ أن ﺗﻌﻠــﻢ أﻧــﻪ ﻳﺤــﻖ ﻟﻨــﺎ‬ ‫ُﻧﻘــﺪر‬ ‫اﺳﺘﺨﺪاﻣﻬﺎ دون أي ﻗﻴﻮد‪.‬‬ ‫ﻧﺤﺘﻔﻆ ﺑﺠﻤﻴﻊ اﻟﺤﻘﻮق اﻟﺘﻲ ﻟﻢ ﻳﺘﻢ ﻣﻨﺤﻬﺎ ﻟﻚ ﺑﺸﻜﻞ ﺻﺮﻳﺢ‪.‬‬ ‫ﻳﺨﻀﻊ اﺳﺘﺨﺪام ﻫﺬه اﻟﺨﺪﻣﺔ ﻟﻬﺬه اﻟﺸﺮوط‪.‬‬
